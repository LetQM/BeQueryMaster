--사용자별 총 구매액
SELECT userID, sum(price*amount) as 총구매액
FROM buyTbl
group by userID

-- 구매액이 1000이상인 고객
/* 
집계함수는 where절 사용x
group by~ having 순서변경x
*/
SELECT userID, sum(price*amount) 
FROM buyTbl
group by userID
having sum(price*amount) > 1000

--각 분류 별로 합계와 총합계
SELECT groupName,sum(price*amount)
FROM buyTbl
group by rollup (groupName)
