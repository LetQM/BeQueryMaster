1번 : 주문량이 많은 아이스크림들 조회하기
 - SELECT FLAVOR FROM (SELECT * FROM FIRST_HALF
   UNION
   SELECT * FROM JULY) A
   GROUP BY A.FLAVOR
   ORDER BY SUM(A.TOTAL_ORDER) DESC
   LIMIT 3

2번 : 저자 별 카테고리 별 매출액 집계하기
 - SELECT a.AUTHOR_ID, a.AUTHOR_NAME, b.CATEGORY, sum(PRICE * SALES) as TOTAL_SALES
   FROM BOOK_SALES bs
   INNER JOIN BOOK b ON b.BOOK_ID = bs.BOOK_ID
   INNER JOIN AUTHOR a ON a.AUTHOR_ID = b.AUTHOR_ID
   WHERE SALES_DATE like '2022-01%'
   GROUP BY AUTHOR_NAME, CATEGORY
   ORDER BY a.AUTHOR_ID, b.CATEGORY DESC
