1번 : 보호소에서 중성화한 동물
 - SELECT ai.ANIMAL_ID, ai.ANIMAL_TYPE, ai.NAME
   FROM ANIMAL_INS ai
   INNER JOIN ANIMAL_OUTS ao
   ON ai.ANIMAL_ID = ao.ANIMAL_ID
   WHERE ai.SEX_UPON_INTAKE != ao.SEX_UPON_OUTCOME
   ORDER BY ai.ANIMAL_ID

2번 : 입양 시각 구하기(2)
- -- 변수 선언
  SET @HOUR := -1;

  SELECT HOUR_TABLE.HOUR, IFNULL(COUNT, 0) as COUNT FROM (SELECT (@HOUR:=@HOUR+1) as HOUR FROM ANIMAL_OUTS limit 24) as HOUR_TABLE
  LEFT JOIN (SELECT hour(DATETIME) as HOUR, count(*) as COUNT FROM ANIMAL_OUTS
  GROUP BY hour(DATETIME)) as RESULT
  ON HOUR_TABLE.HOUR = RESULT.HOUR
  order by HOUR_TABLE.HOUR

  SELECT a.CART_ID FROM CART_PRODUCTS a
  INNER JOIN CART_PRODUCTS b
  ON a.CART_ID = b.CART_ID and a.NAME = 'Yogurt' and b.NAME = 'Milk'
