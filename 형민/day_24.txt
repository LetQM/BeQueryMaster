1번 : 식품분류별 가장 비싼 식품의 정보 조회하기
 - -- 코드를 입력하세요
   SELECT fp.CATEGORY, fp.PRICE as MAX_PRICE, fp.PRODUCT_NAME FROM FOOD_PRODUCT fp
   INNER JOIN
   (SELECT CATEGORY, max(PRICE) as MAX_PRICE FROM FOOD_PRODUCT
   WHERE CATEGORY IN('과자', '국', '김치', '식용유')
   GROUP BY CATEGORY) as GROUP_TABLE
   ON fp.PRICE = GROUP_TABLE.MAX_PRICE AND fp.CATEGORY = GROUP_TABLE.CATEGORY
   ORDER BY MAX_PRICE DESC

2번 : 5월 식품들의 총매출 조회하기
 - -- 코드를 입력하세요
   SELECT fo.PRODUCT_ID, PRODUCT_NAME, PRICE * sum(AMOUNT) as TOTAL_SALES FROM FOOD_ORDER fo
   INNER JOIN FOOD_PRODUCT fp
   ON fo.PRODUCT_ID = fp.PRODUCT_ID
   WHERE DATE_FORMAT(fo.PRODUCE_DATE, '%Y-%m') = '2022-05'
   GROUP BY PRODUCT_ID
   ORDER BY TOTAL_SALES DESC, fo.PRODUCT_ID


3번 : 서울에 위치한 식당 목록 출력하기
 - SELECT ri.REST_ID, REST_NAME, FOOD_TYPE, FAVORITES, ADDRESS, ROUND(AVG(REVIEW_SCORE),2) AS SCORE
   FROM REST_INFO ri
   INNER JOIN REST_REVIEW rr
   ON ri.REST_ID = rr.REST_ID
   WHERE ADDRESS LIKE '서울%'
   GROUP BY REST_ID
   ORDER BY SCORE DESC , FAVORITES DESC
